49  public class StringUtil {
50      /**
51      * Wraps all components in between brackets delimited by ','-signs
52      * @param components to turn into a String
53      * @return a String representation of the input
54      */
55      public static String toString(Object... components) {
56          return toNamedString(null, components);
57      }
58  
59      /**
60      * Wraps all components in between brackets delimited by ','-signs,
61      * appending the class name in front of it.
62      * @param object name to append in front
63      * @param components to turn into a String
64      * @return a String representation of the input
65      */
66      public static String toNamedString(Object object, Object... components) {
67          StringBuilder b = new StringBuilder(components.length * 5 + 2);
68          if (object != null) {
69              b.append(object.getClass().getSimpleName()).append(' ');
70          }
71          b.append('{');
72          for (int i = 0; i < components.length; i++) {
73              if (i != 0) {
74                  b.append(", ");
75              }
76              b.append(components[i]);
77          }
78          b.append('}');
79          return b.toString();
80      }
